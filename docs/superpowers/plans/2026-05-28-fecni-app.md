> **Commit policy (project instruction):** Hold commits until the first working version. A launchable menu-bar app that captures a note to the vault (end of Task 8) is that milestone. Confirm with the user before committing.

> **Build/test environment:** `xcodebuild` and `swift test` self-refuse inside the Claude Code sandbox. Route **all** builds and the app-hosted test target through `./scripts/xcode-build-helper.sh` (the outside-sandbox Hammerspoon daemon) or the `xcode-build` skill. CaptureKit unit tests still run via `swift test --disable-sandbox --package-path Packages/CaptureKit`.

**Goal:** A menu-bar-only macOS app where a global hotkey opens a centered floating editor; jot a note; Esc / click-away saves it as Obsidian Markdown into the configured vault.

**Architecture:** `LSUIElement` app. `fecniApp` hosts a `MenuBarExtra`; an `AppCoordinator` (`@MainActor`, `@Observable`) owns settings, the capture-panel controller, and the hotkey registration. The capture window is a custom `NSPanel` hosting a SwiftUI `CaptureView` whose editor is a macOS 26 rich-text `TextEditor` bound to `AttributedString`. Markdown conversion and file writing are delegated entirely to CaptureKit.

**Tech Stack:** Swift 6.2, SwiftUI + AppKit (`NSPanel`, `NSHostingView`), [`KeyboardShortcuts`](https://github.com/sindresorhus/KeyboardShortcuts) (global hotkey + recorder), `CaptureKit` (local).

- [ ] **Step 2: Write the coordinator**

`AppCoordinator.swift`:
```swift
import SwiftUI
import KeyboardShortcuts
import CaptureKit

@MainActor
@Observable
final class AppCoordinator {
    var settings: CaptureSettings
    private let settingsStore = SettingsStore()
    private lazy var panelController = CapturePanelController(coordinator: self)
    let draftStore = DraftStore()

    init() {
        settings = settingsStore.load()
        KeyboardShortcuts.onKeyUp(for: .openCapture) { [weak self] in
            self?.openCapture()
        }
        recoverDraftIfPresent()
    }

    func openCapture() {
        panelController.present()
    }

    func updateSettings(_ new: CaptureSettings) {
        settings = new
        settingsStore.save(new)
    }

    /// Called by the panel controller when the editor dismisses.
    func commit(markdown: String) {
        defer { draftStore.clear() }
        guard let dir = settings.resolvedDirectory else { return }
        let store = CaptureStore(directory: dir)
        try? store.write(markdown: markdown, at: Date())
    }

    private func recoverDraftIfPresent() {
        guard let recovered = draftStore.load(), let dir = settings.resolvedDirectory else { return }
        try? CaptureStore(directory: dir).write(markdown: recovered, at: Date())
        draftStore.clear()
    }
}
```

`CapturePanelController.swift`:
```swift
import AppKit
import SwiftUI

@MainActor
final class CapturePanelController: NSObject, NSWindowDelegate {
    private let coordinator: AppCoordinator
    private var panel: CapturePanel?
    private var model: CaptureModel?

    init(coordinator: AppCoordinator) {
        self.coordinator = coordinator
    }

    func present() {
        // Fresh empty capture each time.
        let model = CaptureModel(onDismiss: { [weak self] md in self?.finish(markdown: md) })
        self.model = model

        let panel = CapturePanel()
        panel.delegate = self
        panel.contentView = NSHostingView(rootView: CaptureView(model: model))
        panel.center()
        self.panel = panel

        NSApp.activate(ignoringOtherApps: true)
        panel.makeKeyAndOrderFront(nil)
    }

    /// Click-away: when the panel loses key, treat as save & close.
    func windowDidResignKey(_ notification: Notification) {
        model?.requestDismiss()
    }

    private func finish(markdown: String) {
        coordinator.commit(markdown: markdown)
        panel?.delegate = nil
        panel?.close()
        panel = nil
        model = nil
    }
}
```

`CaptureModel` (put in `CaptureView.swift`, Task 5) is `@MainActor @Observable`, holds the `AttributedString`, exposes `requestDismiss()` which calls `onDismiss(MarkdownDocument.markdown(from: text))`.

import SwiftUI
import CaptureKit

@MainActor
@Observable
final class CaptureModel {
    var text = AttributedString()
    var selection = AttributedTextSelection()
    var showFooter = false
    let onDismiss: (String) -> Void

    init(onDismiss: @escaping (String) -> Void) {
        self.onDismiss = onDismiss
    }

    func requestDismiss() {
        onDismiss(MarkdownDocument.markdown(from: text))
    }
}

struct CaptureView: View {
    @Bindable var model: CaptureModel

    var body: some View {
        VStack(spacing: 0) {
            TextEditor(text: $model.text, selection: $model.selection)
                .font(.body)
                .scrollContentBackground(.hidden)
                .padding(24)
                .onChange(of: model.text) { _, newValue in
                    DraftStore().scheduleSave(MarkdownDocument.markdown(from: newValue))
                }

            HStack {
                Spacer()
                Button {
                    model.showFooter.toggle()
                } label: {
                    Image(systemName: "keyboard")
                }
                .buttonStyle(.plain)
                .padding(12)
            }

            if model.showFooter {
                ShortcutsFooter()
            }
        }
        .frame(minWidth: 480, minHeight: 280)
        .background(.regularMaterial)
        .onExitCommand { model.requestDismiss() }   // Esc saves & closes
        .modifier(EditorCommands(model: model))
    }
}

- [ ] **Step 2: Formatting commands**

`EditorFormatting.swift` — a `ViewModifier` attaching the keyboard shortcuts and a formatting definition. Each command mutates the selected range of `model.text`. Implement helpers:
```swift
import SwiftUI
import CaptureKit

struct EditorCommands: ViewModifier {
    @Bindable var model: CaptureModel

    func body(content: Content) -> some View {
        content
            .onKeyPress(.init("b"), phases: .down) { handleIfCommand($0) { Formatter.toggleInline(.stronglyEmphasized, in: model) } }
            .onKeyPress(.init("i"), phases: .down) { handleIfCommand($0) { Formatter.toggleInline(.emphasized, in: model) } }
            .onKeyPress(.init("e"), phases: .down) { handleIfCommand($0) { Formatter.toggleCodeBlock(in: model) } }
            .onKeyPress(.init("1"), phases: .down) { handleIfCommand($0) { Formatter.setHeading(1, in: model) } }
            .onKeyPress(.init("2"), phases: .down) { handleIfCommand($0) { Formatter.setHeading(2, in: model) } }
            .onKeyPress(.init("3"), phases: .down) { handleIfCommand($0) { Formatter.setHeading(3, in: model) } }
            .onKeyPress(.init("l"), phases: .down) { handleIfCommand($0) { Formatter.toggleTaskList(in: model) } }
            .onKeyPress(.init("u"), phases: .down) { handleIfCommandShift($0) { Formatter.toggleList(.unorderedList, in: model) } }
            .onKeyPress(.init("o"), phases: .down) { handleIfCommandShift($0) { Formatter.toggleList(.orderedList, in: model) } }
    }

    private func handleIfCommand(_ press: KeyPress, _ action: () -> Void) -> KeyPress.Result {
        guard press.modifiers == .command else { return .ignored }
        action(); return .handled
    }
    private func handleIfCommandShift(_ press: KeyPress, _ action: () -> Void) -> KeyPress.Result {
        guard press.modifiers == [.command, .shift] else { return .ignored }
        action(); return .handled
    }
}
```
`Formatter` (same file) implements `toggleInline`, `setHeading`, `toggleList`, `toggleTaskList`, `toggleCodeBlock` by mutating `model.text` over `model.selection`'s range(s), setting `inlinePresentationIntent` / `presentationIntent` / `TaskItemAttribute` per CaptureKit's conventions. **Use the exact selection-range API confirmed in the spike.** The correctness contract: after applying a command and dismissing, `MarkdownDocument.markdown(from:)` produces the expected Markdown — assert this in Task 9 smoke tests.

`DraftStore.swift`:
```swift
import Foundation

@MainActor
final class DraftStore {
    private let url: URL
    private var pending: DispatchWorkItem?

    init(fileName: String = "draft.md") {
        let base = FileManager.default.urls(for: .applicationSupportDirectory, in: .userDomainMask)[0]
            .appendingPathComponent("work.miklos.fecni", isDirectory: true)
        try? FileManager.default.createDirectory(at: base, withIntermediateDirectories: true)
        self.url = base.appendingPathComponent(fileName)
    }

    func scheduleSave(_ markdown: String) {
        pending?.cancel()
        let work = DispatchWorkItem { [url] in
            try? markdown.write(to: url, atomically: true, encoding: .utf8)
        }
        pending = work
        DispatchQueue.main.asyncAfter(deadline: .now() + 1.5, execute: work)
    }

    func saveNow(_ markdown: String) {
        pending?.cancel()
        try? markdown.write(to: url, atomically: true, encoding: .utf8)
    }

    func load() -> String? {
        guard let s = try? String(contentsOf: url, encoding: .utf8),
              !s.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty else { return nil }
        return s
    }

    func clear() {
        pending?.cancel()
        try? FileManager.default.removeItem(at: url)
    }
}
```

```swift
import SwiftUI
import KeyboardShortcuts
import CaptureKit

struct SettingsView: View {
    @Bindable var coordinator: AppCoordinator
    @State private var vaults: [ObsidianVault] = VaultLocator.detectedVaults()

    var body: some View {
        Form {
            Section("Vault") {
                Picker("Vault", selection: vaultBinding) {
                    Text("Choose…").tag(String?.none)
                    ForEach(vaults, id: \.path) { v in
                        Text(v.name).tag(Optional(v.path))
                    }
                }
                Button("Choose Folder…") { chooseFolderManually() }

                if let path = coordinator.settings.vaultPath {
                    Picker("Subfolder", selection: subfolderBinding) {
                        Text("(vault root)").tag("")
                        ForEach(VaultLocator.subfolders(ofVault: path), id: \.self) { sub in
                            Text(sub).tag(sub)
                        }
                    }
                }
            }
            Section("Shortcut") {
                KeyboardShortcuts.Recorder("Open capture:", name: .openCapture)
            }
        }
        .formStyle(.grouped)
        .frame(width: 420)
        .padding()
    }

    private var vaultBinding: Binding<String?> {
        Binding(
            get: { coordinator.settings.vaultPath },
            set: { coordinator.updateSettings(CaptureSettings(vaultPath: $0, subfolder: "")) }
        )
    }
    private var subfolderBinding: Binding<String> {
        Binding(
            get: { coordinator.settings.subfolder },
            set: { coordinator.updateSettings(CaptureSettings(vaultPath: coordinator.settings.vaultPath, subfolder: $0)) }
        )
    }

    private func chooseFolderManually() {
        let panel = NSOpenPanel()
        panel.canChooseDirectories = true
        panel.canChooseFiles = false
        if panel.runModal() == .OK, let url = panel.url {
            coordinator.updateSettings(CaptureSettings(vaultPath: url.path, subfolder: ""))
        }
    }
}
```
